106 if (mComponentTypes.find(typeName) != mComponentTypes.end())

108 Logger::_LogError("Attempting to register ComponentType twice.", "vendor/Roc_ECS/include/ComponentManager.h", 107);

112 // Add this component type to the component type map

118 mCreateCompFuncs[typeName] = [=](Entity e){ return this->AddComponent<T>(e); };
